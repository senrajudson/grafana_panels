  <img id="minhaCamera" src="" style="width:100%; max-width:100%; height:auto;" />

// Seleciona o elemento <img> no painel
const img = htmlGraphics.htmlNode.getElementById('minhaCamera');

// Obtém o valor atual da variável do Grafana (ex: Cameras)
const cameraName = htmlGraphics.getTemplateSrv().replace('$Cameras');
// ou se a variável for lowercase:
// const cameraName = htmlGraphics.getTemplateSrv().replace('$camera');

if (cameraName) {
  // Usa template literal para montar o src
  img.src = `http://10.247.179.202:9300/${cameraName}.mjpg`;
} else {
  console.warn('Variável $Cameras está vazia ou não definida.');
}
